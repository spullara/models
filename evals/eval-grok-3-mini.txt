from functools import lru_cache

def solve_grid(grid: list[list[int]]) -> tuple[int, str]:
    if not grid:
        return 0, ""
    N = len(grid)
    
    @lru_cache(None)
    def dp(i: int, j: int, move_count: int) -> tuple[int, str]:
        if i == N-1 and j == N-1:
            return grid[i][j], ""
        
        next_move_num = move_count + 1
        mandatory_X = (next_move_num % 3 == 0)
        
        moves = []
        if i+1 < N and j+1 < N:
            moves.append('X')
        if i+1 < N:
            moves.append('D')
        if j+1 < N:
            moves.append('R')
        
        candidates = []
        for move in moves:
            if mandatory_X and move != 'X':
                continue
            if move == 'X':
                ni, nj = i+1, j+1
            elif move == 'D':
                ni, nj = i+1, j
            elif move == 'R':
                ni, nj = i, j+1
            sub_sum, sub_path = dp(ni, nj, move_count + 1)
            total_sum = grid[i][j] + sub_sum
            candidates.append((total_sum, move + sub_path))
        
        if candidates:
            return max(candidates, key=lambda x: x[0])
        else:
            return float('-inf'), ""
    
    sum_val, path = dp(0, 0, 0)
    return sum_val, path
