                    new_phase = (phase + 1) % 3
                    if new_sum > dp[ni][nj][new_phase]:
                        dp[ni][nj][new_phase] = new_sum
                        pred[ni][nj][new_phase] = (i, j, phase, move_char)
    max_sum = MIN_VAL
    best_phase = -1
    for p in range(3):
        if dp[N - 1][N - 1][p] > max_sum:

            best_phase = p
    if max_sum == MIN_VAL:
        return 0, ""
