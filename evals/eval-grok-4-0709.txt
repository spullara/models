                        prev[ni][nj][next_move_num] = (i, j, k, move)
    best_sum = INF
    best_k = -1
    for kk in range(MAX_K + 1):
        s = dp[N - 1][N - 1][kk]
        if s > best_sum:
            best_sum = s
            best_k = kk
    if best_sum == INF:
        return (0, "")
    # Reconstruct path
    path = []
