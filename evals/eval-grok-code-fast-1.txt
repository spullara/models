def solve_grid(grid: list[list[int]]) -> tuple[int, str]:
    if not grid or not grid[0]:

        required_diagonal = ((m + 1) % 3 == 0)
        possible_moves = []
        if not required_diagonal:

            if i + 1 < N:
                possible_moves.append(('D', i + 1, j))

                possible_moves.append(('X', i + 1, j + 1))

                possible_moves.append(('X', i + 1, j + 1))

                    possible_moves.append(('D', i + 1, j))

            new_sum = current_sum + grid[ni][nj]
            if new_sum > dp[ni][nj][next_m]:
                dp[ni][nj][next_m] = new_sum
                prev[ni][nj][next_m] = (i, j, m, move_char)
                q.append((ni, nj, next_m))
    max_sum = max(dp[N-1][N-1])

    path = []

        path.append(move)

    path.reverse()
    path_string = ''.join(path)
