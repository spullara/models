def solve_grid(grid: list[list[int]]) -> tuple[int, str]:
    if not grid or not grid[0]:
        return 0, ""
    
    n = len(grid)
    INF = float('-inf')
    
    # dp[i][j][k]: max sum to reach (i,j) with k % 3 moves done
    dp = [[[INF] * 3 for _ in range(n)] for _ in range(n)]
    dp[0][0][0] = grid[0][0]
    
    # parent[i][j][k]: path to reach (i,j) with k % 3 moves
    parent = [[[''] * 3 for _ in range(n)] for _ in range(n)]
    
    for i in range(n):
        for j in range(n):
            for k in range(3):
                if dp[i][j][k] == INF:
                    continue
                
                current_moves = k + 3 * ((i + j) // 2)  # approximate total moves
                is_diag_required = (current_moves + 1) % 3 == 0
                
                # Possible moves: right, down, diagonal
                moves = []
                if j + 1 < n:
                    moves.append(('R', i, j + 1))
                if i + 1 < n:
                    moves.append(('D', i + 1, j))
                if i + 1 < n and j + 1 < n:
                    moves.append(('X', i + 1, j + 1))
                
                for move, ni, nj in moves:
                    if is_diag_required and move != 'X':
                        continue  # Skip non-diagonal if diagonal required
                    
                    new_k = (k + 1) % 3
                    new_sum = dp[i][j][k] + grid[ni][nj]
                    
                    if new_sum > dp[ni][nj][new_k]:
                        dp[ni][nj][new_k] = new_sum
                        if parent[i][j][k]:
                            parent[ni][nj][new_k] = parent[i][j][k] + move
                        else:
                            parent[ni][nj][new_k] = move
    
    # Find max sum at bottom-right
    max_sum = INF
    best_k = -1
    for k in range(3):
        if dp[n-1][n-1][k] > max_sum:
            max_sum = dp[n-1][n-1][k]
            best_k = k
    
    if max_sum == INF:
        return 0, ""
    
    path = parent[n-1][n-1][best_k]
    return int(max_sum), path
