    if not grid:
        return 0, ""
    N = len(grid)
    if N == 1:
        return grid[0][0], ""

    pred = [[[(None, None, None) for _ in range(3)] for _ in range(N)] for _ in range(N)]
    
    DP[0][0][0] = grid[0][0]
    
    for row in range(N):
        for col in range(N):
            if row == 0 and col == 0:

                predecessors.append((row, col-1, 'R'))

                predecessors.append((row-1, col, 'D'))

                predecessors.append((row-1, col-1, 'X'))

                                DP[row][col][new_mod] = potential_sum
                                pred[row][col][new_mod] = (prev_row, prev_col, prev_mod)

    end_mod = max(range(3), key=lambda m: DP[N-1][N-1][m])
