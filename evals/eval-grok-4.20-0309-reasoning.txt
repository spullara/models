                        continue

                        dp[ni][nj][new_m] = new_sum
                        prev[ni][nj][new_m] = (i, j, m, char)
    max_sum = -INF
    best_m = -1
    for m_val in range(MAX_M):
        if dp[n-1][n-1][m_val] > max_sum:

            best_m = m_val
    if best_m == -1 or max_sum == -INF:
