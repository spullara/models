        return (0, "")

        return (grid[0][0], "")

                step = moves + 1
                must_diag = (step % 3 == 0)
                
                candidates = []
                if j + 1 < n:
                    candidates.append(('R', i, j + 1))

                    candidates.append(('D', i + 1, j))
                if i + 1 < n and j + 1 < n:
                    candidates.append(('X', i + 1, j + 1))
                
                for dir_char, ni, nj in candidates:
                    if must_diag and dir_char != 'X' and any(d == 'X' for d,_,_ in candidates if (d=='X' and (i+1<n and j+1<n))):
                        continue

                    new_path = path + dir_char
                    new_moves = moves + 1
                    
                    if new_moves not in dp[ni][nj] or new_sum > dp[ni][nj][new_moves][0]:
                        dp[ni][nj][new_moves] = (new_sum, new_path)

    best_path = ""
