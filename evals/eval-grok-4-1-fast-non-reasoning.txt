def solve_grid(grid: list[list[int]]) -> tuple[int, str]:
    if not grid or not grid[0]:
        return 0, ""
    
    n = len(grid)
    INF = float('-inf')
    
    # dp[i][j][k]: max sum to reach (i,j) with k moves made
    dp = [[[INF] * (n*n + 1) for _ in range(n)] for _ in range(n)]
    dp[0][0][0] = grid[0][0]
    
    # parent[i][j][k] stores the previous position and move for reconstruction
    parent = [[[(None, None) for _ in range(n*n + 1)] for _ in range(n)] for _ in range(n)]
    
    for i in range(n):
        for j in range(n):
            for k in range(n*n + 1):
                if dp[i][j][k] == INF:
                    continue

                diagonal_required = (k + 1) % 3 == 0

                moves = []
                if j + 1 < n:  # Right
                    moves.append(('R', i, j + 1))
                if i + 1 < n:  # Down
                    moves.append(('D', i + 1, j))
                if i + 1 < n and j + 1 < n:  # Diagonal
                    moves.append(('X', i + 1, j + 1))

                    if diagonal_required and move_type != 'X':
                        continue
                    
                    new_sum = dp[i][j][k] + grid[ni][nj]
                    if new_sum > dp[ni][nj][k + 1]:
                        dp[ni][nj][k + 1] = new_sum
                        parent[ni][nj][k + 1] = ((i, j, k), move_type)
    
    # Find max sum at bottom-right
    max_sum = INF
    best_k = -1
    for k in range(n*n + 1):
        if dp[n-1][n-1][k] > max_sum:
            max_sum = dp[n-1][n-1][k]
            best_k = k
    
    if max_sum == INF:
        return 0, ""

    path = []
    current = (n-1, n-1, best_k)

        prev, move = parent[current[0]][current[1]][current[2]]

    path.reverse()
