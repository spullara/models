    if not grid or not grid[0]:

                    moves.append(('D', i + 1, j))

                    if not diag_possible:
                        # Use R or D if diagonal impossible

                        valid_moves = [('X', moves[2][1], moves[2][2])]  # Only diagonal
                else:
                    valid_moves = moves
                
                for move_char, ni, nj in valid_moves:

                    new_path = path + move_char
                    new_k = (k + 1) % 3
