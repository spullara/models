import sys

def solve_grid(grid: list[list[int]]) -> tuple[int, str]:

        if i == N - 1 and j == N - 1:
            return grid[i][j], ""

        must_diag = (next_move_num % 3 == 0)
        
        possible_moves = []
        if must_diag:
            if i + 1 < N and j + 1 < N:
                possible_moves = ['X']
            else:
                possible_moves = ['R', 'D'] if j + 1 < N else (['D'] if i + 1 < N else [])
        else:
            possible_moves = []
            if j + 1 < N:
                possible_moves.append('R')

                possible_moves.append('D')

                possible_moves.append('X')
        
        max_sum = -sys.maxsize
        best_path = ""
        
        for move in possible_moves:
            ni, nj = i, j

                nj += 1
