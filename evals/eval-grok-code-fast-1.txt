from collections import deque

def solve_grid(grid: list[list[int]]) -> tuple[int, str]:
    N = len(grid)

                possible_moves.append(('X', i + 1, j + 1))

                    possible_moves.append(('R', i, j + 1))
                if i + 1 < N:
                    possible_moves.append(('D', i + 1, j))

            if j + 1 < N:
                possible_moves.append(('R', i, j + 1))
            if i + 1 < N:
                possible_moves.append(('D', i + 1, j))

                possible_moves.append(('X', i + 1, j + 1))

            new_k = (k + 1) % 3
            new_sum = current_sum + grid[ni][nj]
            if new_sum > dp[ni][nj][new_k]:
                dp[ni][nj][new_k] = new_sum
                prev[ni][nj][new_k] = (i, j, k, move)
                q.append((ni, nj, new_k))
    
    max_sum = max(dp[N-1][N-1])
    if max_sum == INF:
        return (0, "")
    
    for k in range(3):
        if dp[N-1][N-1][k] == max_sum:
            best_k = k
            break
    
    path = []
    current_i, current_j, current_k = N-1, N-1, best_k
    while current_i != 0 or current_j != 0 or current_k != 0:
