    max_sum = -1
    best_path = ""

        required_diagonal = (next_move_num % 3 == 0)

        possible_moves = []
        if required_diagonal:
            if i + 1 < n and j + 1 < n:
                possible_moves.append('X')
            else:
                possible_moves.extend(['R', 'D'])

        if not required_diagonal:
            if j + 1 < n:
                possible_moves.append('R')

                possible_moves.append('D')
            if i + 1 < n and j + 1 < n:
                possible_moves.append('X')
