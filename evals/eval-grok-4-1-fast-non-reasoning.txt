    if not grid or not grid[0]:
        return 0, ""
    
    n = len(grid)
    INF = float('-inf')
    
    # dp[i][j][k]: max sum to reach (i,j) with k moves made
    dp = [[[INF] * (n*n + 1) for _ in range(n)] for _ in range(n)]
    dp[0][0][0] = grid[0][0]
    
    # parent[i][j][k] stores previous position and move for path reconstruction
    parent = [[[(None, None) for _ in range(n*n + 1)] for _ in range(n)] for _ in range(n)]

                    continue

                is_diag_required = (total_moves + 1) % 3 == 0
                
                # Possible moves: R, D, X
                moves = []
                if j + 1 < n:
                    moves.append(('R', i, j + 1))
                if i + 1 < n:
                    moves.append(('D', i + 1, j))
                if i + 1 < n and j + 1 < n:
                    moves.append(('X', i + 1, j + 1))

                    if is_diag_required and move_type != 'X' and ('X', ni, nj) not in moves:
                        continue
                    
                    new_sum = current_sum + grid[ni][nj]
                    new_moves = total_moves + 1
                    
                    if new_sum > dp[ni][nj][new_moves]:
                        dp[ni][nj][new_moves] = new_sum
                        parent[ni][nj][new_moves] = ((i, j, total_moves), move_type)
    
    # Find max sum at bottom-right
    max_sum = INF
    best_moves = -1
    for k in range(n*n + 1):
        if dp[n-1][n-1][k] > max_sum:
            max_sum = dp[n-1][n-1][k]
            best_moves = k
    
    if max_sum == INF:
        return 0, ""
    
    # Reconstruct path
    path = []
    i, j, moves = n-1, n-1, best_moves
    while i > 0 or j > 0:
        prev_pos, move_type = parent[i][j][moves]
        path.append(move_type)

    path.reverse()
    return max_sum, ''.join(path)
