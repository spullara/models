from functools import lru_cache
from typing import List, Tuple

def solve_grid(grid: List[List[int]]) -> Tuple[int, str]:
    if not grid or not grid[0]:
        return (0, "")
    N = len(grid)
    
    @lru_cache(None)
    def recurse(i: int, j: int, move_num: int) -> Tuple[int, str]:
        if i == N - 1 and j == N - 1:
            return (0, "")
        
        possible_moves = []
        required_diagonal = (move_num + 1) % 3 == 0
        
        # Check if X is possible
        can_x = i < N - 1 and j < N - 1
        
        if can_x:
            possible_moves.append(('X', i + 1, j + 1))
        if not required_diagonal or not can_x:
            if j < N - 1:
                possible_moves.append(('R', i, j + 1))
            if i < N - 1:
                possible_moves.append(('D', i + 1, j))
        
        max_sum = float('-inf')
        best_path = ""
        
        for move, ni, nj in possible_moves:
            add_sum, path = recurse(ni, nj, move_num + 1)
            current_sum = grid[ni][nj] + add_sum
            if current_sum > max_sum:
                max_sum = current_sum
                best_path = move + path
        
        return (max_sum, best_path)
    
    initial_sum = grid[0][0]
    additional_sum, path = recurse(0, 0, 0)
    return (initial_sum + additional_sum, path)
