    if not grid or not grid[0]:
        return 0, ""

        raise ValueError("Grid must be square")

    dp = [[[(-float('inf'), "") for _ in range(3)] for _ in range(n)] for _ in range(n)]

                    continue

                next_moves = k + 1
                
                # Possible moves
                moves = []
                if j + 1 < n:
                    moves.append(('R', i, j + 1, grid[i][j + 1]))
                if i + 1 < n:
                    moves.append(('D', i + 1, j, grid[i + 1][j]))
                if i + 1 < n and j + 1 < n:
                    moves.append(('X', i + 1, j + 1, grid[i + 1][j + 1]))
                
                for move_char, ni, nj, value in moves:
                    next_k = next_moves % 3
                    is_diag_required = (next_moves % 3 == 0)
                    
                    if is_diag_required and move_char != 'X':
                        # Check if diagonal is possible
                        has_diag = any(m[0] == 'X' for m in moves)
                        if has_diag:
                            continue  # Skip non-diagonal if diagonal is possible
                    
                    new_sum = current_sum + value
                    new_path = path + move_char
                    
                    if new_sum > dp[ni][nj][next_k][0]:
                        dp[ni][nj][next_k] = (new_sum, new_path)
