    if not grid or not grid[0]:
        return 0, ""
    N = len(grid)
    INF = float('-inf')
    dp = [[[INF] * 3 for _ in range(N)] for _ in range(N)]

                            possible_moves.append((i + 1, j + 1, 'X'))
                        else:

                                possible_moves.append((i, j + 1, 'R'))
                            if i + 1 < N:
                                possible_moves.append((i + 1, j, 'D'))

                        # Can try all

                            possible_moves.append((i, j + 1, 'R'))
                        if i + 1 < N:
                            possible_moves.append((i + 1, j, 'D'))

                            possible_moves.append((i + 1, j + 1, 'X'))

                        new_sum = dp[i][j][k] + grid[ni][nj]
                        new_k = (k + 1) % 3
                        if new_sum > dp[ni][nj][new_k]:
                            dp[ni][nj][new_k] = new_sum
                            prev[ni][nj][new_k] = (i, j, k, move)
    
    # Find the max sum at bottom-right
    max_sum = INF
    best_k = -1
    for k in range(3):
        if dp[N-1][N-1][k] > max_sum:
            max_sum = dp[N-1][N-1][k]
            best_k = k
    
    if max_sum == INF:
        return 0, ""  # No path, though unlikely
    
    # Reconstruct path
    path = []
    i, j, k = N-1, N-1, best_k
    while i != 0 or j != 0 or k != 0:
        pi, pj, pk, move = prev[i][j][k]
        path.append(move)
        i, j, k = pi, pj, pk
    path.reverse()
    path_string = ''.join(path)
    return max_sum, path_string
