    # move_count_mod_3 is the number of moves taken so far modulo 3.
    # We initialize the start state at (0, 0) which has 0 moves taken.
    # Thus, the next move will be the 1st move.
    dp = [[[] for _ in range(N)] for _ in range(N)]

                    # Let's allow all valid moves.
