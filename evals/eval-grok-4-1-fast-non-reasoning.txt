    if not grid or not grid[0]:
        return 0, ""
    n = len(grid)
    INF = float('-inf')

    dp[0][0][0] = (grid[0][0], "")

                    ni, nj = i + di, j + dj

                            continue
                        new_sum = dp[i][j][k_mod][0] + grid[ni][nj]
                        new_path = dp[i][j][k_mod][1] + move
                        new_k_mod = (moves_made + 1) % 3

                            dp[ni][nj][new_k_mod] = (new_sum, new_path)
