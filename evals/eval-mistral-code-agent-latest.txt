def solve_grid(grid: list[list[int]]) -> tuple[int, str]:
    n = len(grid)
    if n == 0:
        return (0, "")

    from heapq import heappop, heappush

    heap = []
    heappush(heap, (-grid[0][0], 0, 0, 0, ""))

    visited = {}

    while heap:
        neg_sum, x, y, moves, path = heappop(heap)
        current_sum = -neg_sum

        if (x, y, moves) in visited and visited[(x, y, moves)] >= current_sum:
            continue

        visited[(x, y, moves)] = current_sum

        if x == n - 1 and y == n - 1:
