    N = len(grid)

            return memo[(i, j, mod_next)]

            memo[(i, j, mod_next)] = (grid[i][j], "")
            return memo[(i, j, mod_next)]

        best_path = ""
        
        for next_i, next_j in possible:
            new_mod = (mod_next + 1) % 3
            sub_sum, sub_path = dp(next_i, next_j, new_mod)
            total_sum_for_this = grid[i][j] + sub_sum

                max_sum = total_sum_for_this
                if next_i == i and next_j == j+1:
