def solve_grid(grid: list[list[int]]) -> tuple[int, str]:
    N = len(grid)
    
    @lru_cache(None)
    def dp(i: int, j: int, mod: int) -> tuple[int, str]:
        if i == N - 1 and j == N - 1:
            return grid[i][j], ""
        
        possible = []
        must_X = (mod == 2)
        if must_X:
            if i + 1 < N and j + 1 < N:
                possible.append(('X', i + 1, j + 1))
            else:
                if j + 1 < N:
                    possible.append(('R', i, j + 1))
                if i + 1 < N:
                    possible.append(('D', i + 1, j))
        else:
            if j + 1 < N:
                possible.append(('R', i, j + 1))
            if i + 1 < N:
                possible.append(('D', i + 1, j))
            if i + 1 < N and j + 1 < N:
                possible.append(('X', i + 1, j + 1))
        
        best_sum = float('-inf')
        best_path = ""
        for move, ni, nj in possible:
            sub_sum, sub_path = dp(ni, nj, (mod + 1) % 3)
            total_sum = grid[i][j] + sub_sum
            if total_sum > best_sum:
                best_sum = total_sum
                best_path = move + sub_path
        return best_sum, best_path
